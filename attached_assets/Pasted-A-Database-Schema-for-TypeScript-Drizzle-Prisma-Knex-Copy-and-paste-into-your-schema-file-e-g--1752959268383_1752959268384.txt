// --- INVENTORY ITEM ---
export interface InventoryItem {
  id: number;
  name: string;
  quantity: number; // current amount in stock
  unit: string; // e.g. "oz", "each", "box"
  parLevel: number; // restock threshold
  archived: boolean;
  createdAt: Date;
  updatedAt: Date;
}
export type InsertInventoryItem = Omit<InventoryItem, "id" | "createdAt" | "updatedAt" | "archived">;

// --- INVENTORY ADJUSTMENT LOG ---
export interface InventoryAdjustment {
  id: number;
  inventoryItemId: number;
  adjustment: number; // positive (add), negative (remove)
  reason: string; // e.g., "manual sale", "waste", "delivery", "comp"
  note: string;
  userId: number;
  createdAt: Date;
}
export type InsertInventoryAdjustment = Omit<InventoryAdjustment, "id" | "createdAt">;

// --- MANUAL ENTRY FLAG ON ORDERS ---
export interface Order {
  // ...existing fields

// Get all inventory (not archived)
async getInventory() {
  return await db.select().from(inventoryItems).where(eq(inventoryItems.archived, false));
}

// Get low-stock inventory
async getInventoryLowStock() {
  return await db.select().from(inventoryItems)
    .where(and(eq(inventoryItems.archived, false), lte(inventoryItems.quantity, inventoryItems.parLevel)));
}

// Create item
async createInventoryItem(item) {
  const [created] = await db.insert(inventoryItems).values({ ...item, archived: false, createdAt: new Date(), updatedAt: new Date() }).returning();
  return created;
}

// Update item
async updateInventoryItem(id, item) {
  const [updated] = await db.update(inventoryItems)
    .set({ ...item, updatedAt: new Date() })
    .where(eq(inventoryItems.id, id))
    .returning();
  return updated;
}

// Archive (soft-delete) item
async archiveInventoryItem(id) {
  await db.update(inventoryItems)
    .set({ archived: true, updatedAt: new Date() })
    .where(eq(inventoryItems.id, id));
}

// Adjust inventory & log
async adjustInventoryItem({ inventoryItemId, adjustment, reason, note, userId }) {
  // Insert log entry
  await db.insert(inventoryAdjustments).values({
    inventoryItemId, adjustment, reason, note, userId, createdAt: new Date()
  });
  // Update stock
  const [updated] = await db.update(inventoryItems)
    .set({ quantity: sql`${inventoryItems.quantity} + ${adjustment}`, updatedAt: new Date() })
    .where(eq(inventoryItems.id, inventoryItemId))
    .returning();
  return updated;
}

// Get log
async getInventoryAdjustments() {
  return await db.select().from(inventoryAdjustments);
}
async getInventoryAdjustmentsByItem(itemId) {
  return await db.select().from(inventoryAdjustments)
    .where(eq(inventoryAdjustments.inventoryItemId, itemId));
}

  quantity: z.number().optional(),
  unit: z.string().optional(),
  parLevel: z.number().optional()

  res.json(successResponse(updated, "Inventory item updated"));
}));

// Archive
app.delete("/api/inventory/:id", authenticateAdmin, asyncHandler(async (req, res) => {
  await storage.archiveInventoryItem(parseInt(req.params.id));
  res.json(successResponse(null, "Inventory item archived"));
}));

// Adjust (add/remove for delivery, waste, etc.)
app.post("/api/inventory/:id/adjust", authenticateAdmin, validateBody(z.object({
  adjustment: z.number(),
  reason: z.string(),
  note: z.string().optional()
})), asyncHandler(async (req, res) => {
  const userId = req.user.id;
  const updated = await storage.adjustInventoryItem({
    inventoryItemId: parseInt(req.params.id),
    adjustment: req.body.adjustment,
    reason: req.body.reason,
    note: req.body.note || "",
    userId
  });
  res.json(successResponse(updated, "Inventory adjusted"));
}));

// Log
app.get("/api/inventory-log", authenticateAdmin, asyncHandler(async (req, res) => {
  const log = await storage.getInventoryAdjustments();
  res.json(successResponse(log));
}));

// --- MANUAL TICKET ENTRY ---

app.post("/api/manual-ticket", authenticateAdmin, validateBody(z.object({
  items: z.array(z.object({
    inventoryItemId: z.number(),
    quantity: z.number()
  })),
  totalAmount: z.number(),
  customerName: z.string().optional(),
  note: z.string().optional()
})), asyncHandler(async (req, res) => {
  // 1. Create an order flagged as manual
  const order = await storage.createOrder({
    ...req.body,
    manualEntry: true,
    createdAt: new Date()
  });
  // 2. Deduct inventory/log for each item
  for (const item of req.body.items) {
    await storage.adjustInventoryItem({
      inventoryItemId: item.inventoryItemId,
      adjustment: -item.quantity,
      reason: "manual ticket",
      note: req.body.note || "",
      userId: req.user.id
    });
  }
  res.json(successResponse(order, "Manual ticket entered, inventory updated"));
}));

// AdminInventory.tsx

import React, { useEffect, useState } from "react";
import axios from "axios";

export default function AdminInventory() {
  const [items, setItems] = useState([]);
  const [lowStock, setLowStock] = useState([]);
  const [log, setLog] = useState([]);
  const [adjustId, setAdjustId] = useState(null);
  const [adjustValue, setAdjustValue] = useState(0);
  const [adjustReason, setAdjustReason] = useState("");
  const [adjustNote, setAdjustNote] = useState("");

  useEffect(() => {
    fetchInventory();
    fetchLog();
  }, []);

  async function fetchInventory() {
    const { data } = await axios.get("/api/inventory");
    setItems(data.data.inventory);
    setLowStock(data.data.lowStock);
  }

  async function fetchLog() {
    const { data } = await axios.get("/api/inventory-log");
    setLog(data.data);
  }

  async function handleAdjust(id) {
    await axios.post(`/api/inventory/${id}/adjust`, {
      adjustment: adjustValue,
      reason: adjustReason,
      note: adjustNote
    });
    fetchInventory();
    setAdjustId(null);
  }

  return (
    <div>
      <h2>Inventory Management</h2>
      <table>
        <thead>
          <tr>
            <th>Name</th><th>Qty</th><th>Unit</th><th>Par</th><th>Status</th><th>Action</th>
          </tr>
        </thead>
        <tbody>
          {items.map(item => (
            <tr key={item.id} style={{ background: item.quantity <= item.parLevel ? "#ffbaba" : undefined }}>
              <td>{item.name}</td>
              <td>{item.quantity}</td>
              <td>{item.unit}</td>
              <td>{item.parLevel}</td>
              <td>
                {item.quantity <= item.parLevel ? <b style={{ color: "red" }}>Low</b> : "OK"}
              </td>
              <td>
                <button onClick={() => setAdjustId(item.id)}>Adjust</button>
              </td>
            </tr>
          ))}
        </tbody>
      </table>
      {adjustId && (
        <div>
          <h3>Adjust Inventory</h3>
          <input type="number" value={adjustValue} onChange={e => setAdjustValue(Number(e.target.value))} placeholder="Amount (+/-)" />
          <input type="text" value={adjustReason} onChange={e => setAdjustReason(e.target.value)} placeholder="Reason" />
          <input type="text" value={adjustNote} onChange={e => setAdjustNote(e.target.value)} placeholder="Note (optional)" />
          <button onClick={() => handleAdjust(adjustId)}>Submit</button>
          <button onClick={() => setAdjustId(null)}>Cancel</button>
        </div>
      )}
      <h3>Adjustment Log</h3>
      <table>
        <thead>
          <tr><th>Item</th><th>Change</th><th>Reason</th><th>Note</th><th>User</th><th>Date</th></tr>
        </thead>
        <tbody>
          {log.map(entry => (
            <tr key={entry.id}>
              <td>{entry.inventoryItemId}</td>
              <td>{entry.adjustment}</td>
              <td>{entry.reason}</td>
              <td>{entry.note}</td>
              <td>{entry.userId}</td>
              <td>{(new Date(entry.createdAt)).toLocaleString()}</td>
            </tr>
          ))}
        </tbody>
      </table>
    </div>
  );
}

// ManualTicketEntry.tsx

import React, { useEffect, useState } from "react";
import axios from "axios";

export default function ManualTicketEntry() {
  const [inventory, setInventory] = useState([]);
  const [ticketItems, setTicketItems] = useState([]);
  const [customerName, setCustomerName] = useState("");
  const [totalAmount, setTotalAmount] = useState(0);
  const [note, setNote] = useState("");

  useEffect(() => {
    axios.get("/api/inventory").then(({ data }) => setInventory(data.data.inventory));
  }, []);

  function addItem(itemId) {
    setTicketItems([...ticketItems, { inventoryItemId: itemId, quantity: 1 }]);
  }
  function updateQty(index, qty) {
    const next = [...ticketItems];
    next[index].quantity = qty;
    setTicketItems(next);
  }
  async function handleSubmit() {
    await axios.post("/api/manual-ticket", {
      items: ticketItems,
      customerName,
      totalAmount,
      note
    });
    setTicketItems([]);
    setCustomerName("");
    setTotalAmount(0);
    setNote("");
    alert("Manual ticket submitted and inventory updated!");
  }

  return (
    <div>
      <h2>Manual Ticket Entry</h2>
      <input type="text" value={customerName} onChange={e => setCustomerName(e.target.value)} placeholder="Customer Name (optional)" />
      <input type="number" value={totalAmount} onChange={e => setTotalAmount(Number(e.target.value))} placeholder="Total Amount" />
      <input type="text" value={note} onChange={e => setNote(e.target.value)} placeholder="Note (optional)" />
      <h3>Add Items:</h3>
      <ul>
        {inventory.map(item => (
          <li key={item.id}>
            {item.name} ({item.quantity} {item.unit} in stock)
            <button onClick={() => addItem(item.id)}>Add</button>
          </li>
        ))}
      </ul>
      <h3>Current Ticket:</h3>
      <ul>
        {ticketItems.map((item, i) => (
          <li key={i}>
            {inventory.find(x => x.id === item.inventoryItemId)?.name} -
            <input
              type="number"
              value={item.quantity}
              min={1}
              onChange={e => updateQty(i, Number(e.target.value))}
            />
          </li>
        ))}
      </ul>
      <button onClick={handleSubmit}>Submit Ticket</button>
    </div>
  );
}

function exportCSV() {
  const csvRows = [
    ["Item ID", "Change", "Reason", "Note", "User", "Date"],
    ...log.map(e =>
      [e.inventoryItemId, e.adjustment, e.reason, e.note, e.userId, (new Date(e.createdAt)).toLocaleString()]
    )
  ];
  const csvString = csvRows.map(row => row.join(",")).join("\n");
  const blob = new Blob([csvString], { type: "text/csv" });
  const url = URL.createObjectURL(blob);
  const a = document.createElement("a");
  a.href = url;
  a.download = "inventory-log.csv";
  a.click();
}
